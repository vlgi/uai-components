# Emoji Picker

## CSS customization

| CSS Property                                     | Default                    | Descripiton                      |
|--------------------------------------------------|----------------------------|----------------------------------|
| --szot-emojis-picker-background-color            | white                      | Container background color       |
| --szot-emojis-picker-border-radius               | var(--theme-small-shape)   | Container border radius          |
| --szot-emojis-picker-box-shadow                  | var(--theme-medium-shadow) | Container box shadow             |
| --szot-emojis-picker-height                      | 315px                      | Container height                 |
| --szot-emojis-picker-width                       | 315px                      | Container width                  |
| --szot-emojis-picker-category-title-border-color | var(--theme-light)         | Category title border color      |
| --szot-emojis-picker-category-title-font-color   | var(--theme-dark)          | Category title font color        |
| --szot-emojis-picker-emoji-background-color      | transparent                | Emoji button background color    |
| --szot-emojis-picker-tabs-border-color           | var(--theme-light)         | Category tab button border color |
| --szot-emojis-picker-tabs-icon-color             | var(--theme-dark)         | Category tab icon color          |

## Usage example

```html
<script lang="ts">
  import { onMount } from "svelte";
  import EmojiPicker from "szot-ui-experimental";
  import Input from "szot-ui-experimental";

  let targetEl = null;
  let msg = "";
  let opened = true;

  onMount(() => {
    targetEl = document.getElementById("input-msg");
  });

  function onTypeMsg(e) {
    msg = (e.target as HTMLInputElement).value;
  }

  function openEmojisPicker() {
    opened = true;
  }

  $: selectedEmoji = "";
  $: msg = `${msg}${selectedEmoji as string}`;
</script>

<div class="container">
  <div class="input-target-container">
    <Input
      on:input={onTypeMsg}
      on:click={openEmojisPicker}
      icon="emoticon-happy"
      iconClick={true}
      id="input-msg"
      inputAttributes={{ autocomplete: "off" }}
      label="Message"
      name="message"
      type="text"
      value={msg}
      --szot-input-margin-bottom="0"
      --szot-input-margin-top="0"
    />
  </div>

  <EmojiPicker
    bind:opened
    bind:selected={selectedEmoji}
    localStorageId="szot-ui-emoji-picker-example"
    {targetEl}
    --szot-emojis-picker-background-color="white"
    --szot-emojis-picker-border-radius="25px"
    --szot-emojis-picker-height="350px"
    --szot-emojis-picker-width="320px"
    --szot-emojis-picker-category-title-border-color="#ddd"
    --szot-emojis-picker-category-title-font-color="#444"
    --szot-emojis-picker-emoji-background-color="white"
    --szot-emojis-picker-tabs-border-color="#eee"
    --szot-emojis-picker-tabs-icon-color="#777"
  />
</div>

<style lang="scss">
  .container {
    padding: 20px;
    display: flex;
    flex-direction: column;
    align-items: flex-end;
    background: #eee;
    width: 100%;
    height: 600px;
    max-height: 90vh;

    .input-target-container {
      width: 100%;
      margin-bottom: 5px;
    }
  }
</style>
```
